\section{PIC equal stiff system}
\begin{equation}
 \dot{\textbf{y}} = \left(
        \begin{array}{cc}
           -1 & 0.001\\
           0.001  & -0.001
        \end{array}
     \right) \textbf{y} + \left(
        \begin{array}{c}
           0 \\
           f_{Force}(t)
        \end{array}
     \right),
\label{ExtForceStiff}
\end{equation}
with a decreasing force,
\begin{equation}
 f_{Force}(t) = e^{-t} + e^{-0.001t},
\end{equation}

method: Slowest first 1 (with reevaluatio)\\
